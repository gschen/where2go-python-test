        \node (dec1) [decision, below of=pro1, yshift=-0.5cm] {Decision 1};
        \node (pro2a) [process, below of=dec1, yshift=-1.5cm] {Process 2a long long long text long long long text};

        \node (stop) [startstop, below of=out1] {Stop};
